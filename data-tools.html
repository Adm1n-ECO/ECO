Step 3
Auto-derive relationships from structure
Once parent/partner data is correct, run this SQL in Supabase SQL Editor. It clears all existing relationship entries for lightning-001 and regenerates them from the structural fields (parent_id, co_parent_id, partner_id). Relationship titles will be accurate automatically.
-- Step 3: Auto-derive relationships for lightning-001 -- Run in Supabase SQL Editor after correcting parent/partner data -- SAFE: only clears source='derived' rows. Manual corrections are NEVER touched. -- 1. Clear only derived relationships (manual entries are preserved) DELETE FROM relationships WHERE network_id = 'lightning-001' AND source = 'derived'; -- 2. Parent → Child INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, parent_id, user_id, 'parent', 'derived' FROM users WHERE parent_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 3. Child → Parent INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, user_id, parent_id, 'child', 'derived' FROM users WHERE parent_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 4. Co-parent → Child INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, co_parent_id, user_id, 'parent', 'derived' FROM users WHERE co_parent_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 5. Child → Co-parent INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, user_id, co_parent_id, 'child', 'derived' FROM users WHERE co_parent_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 6. Partner (bidirectional) INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, user_id, partner_id, 'spouse', 'derived' FROM users WHERE partner_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT network_id, partner_id, user_id, 'spouse', 'derived' FROM users WHERE partner_id IS NOT NULL AND network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 7. Siblings (share same parent_id) INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT a.network_id, a.user_id, b.user_id, 'sibling', 'derived' FROM users a JOIN users b ON a.parent_id = b.parent_id AND a.user_id != b.user_id AND a.parent_id IS NOT NULL AND a.network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 8. Grandparent → Grandchild INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT p.network_id, p.parent_id, c.user_id, 'grandparent', 'derived' FROM users c JOIN users p ON c.parent_id = p.user_id WHERE p.parent_id IS NOT NULL AND c.network_id = 'lightning-001' ON CONFLICT DO NOTHING; INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT p.network_id, c.user_id, p.parent_id, 'grandchild', 'derived' FROM users c JOIN users p ON c.parent_id = p.user_id WHERE p.parent_id IS NOT NULL AND c.network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 9. Aunt/Uncle ↔ Niece/Nephew (parent's sibling) INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT child.network_id, sibling.user_id, child.user_id, 'aunt_uncle', 'derived' FROM users child JOIN users parent ON child.parent_id = parent.user_id JOIN users sibling ON sibling.parent_id = parent.parent_id AND sibling.user_id != parent.user_id WHERE child.network_id = 'lightning-001' ON CONFLICT DO NOTHING; INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT child.network_id, child.user_id, sibling.user_id, 'niece_nephew', 'derived' FROM users child JOIN users parent ON child.parent_id = parent.user_id JOIN users sibling ON sibling.parent_id = parent.parent_id AND sibling.user_id != parent.user_id WHERE child.network_id = 'lightning-001' ON CONFLICT DO NOTHING; -- 10. Cousins (share grandparent) INSERT INTO relationships (network_id, subject_id, target_id, relation_type, source) SELECT a.network_id, a.user_id, b.user_id, 'cousin', 'derived' FROM users a JOIN users pa ON a.parent_id = pa.user_id JOIN users pb ON pb.parent_id = pa.parent_id AND pb.user_id != pa.user_id JOIN users b ON b.parent_id = pb.user_id AND b.user_id != a.user_id WHERE a.network_id = 'lightning-001' ON CONFLICT DO NOTHING; SELECT 'Done. Relationships regenerated from structure.' as result;
